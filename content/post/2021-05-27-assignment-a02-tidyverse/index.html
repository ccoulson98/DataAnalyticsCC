Determining The % of top 10 finishes for players who played 20 or more events

golf2017UP <- golf2017 %>%
  group_by(NUMBER_OF_TOP_Tens) %>%
 filter(NUMBER_OF_TOP_Tens >= 3) %>%
  mutate(TOP10_percent = NUMBER_OF_TOP_Tens/EVENTS_PLAYED)
